function doGet(e) { const MY_SECRET_KEY = PropertiesService.getScriptProperties().getProperty('MY_KEY'); const WEATHER_API_KEY = PropertiesService.getScriptProperties().getProperty('WEATHER_KEY'); if (!MY_SECRET_KEY) { return createJsonResponse({error: "Server secret key is not set."}); } if (!e || !e.parameter) { return createJsonResponse({error: "Invalid request format."}); } if (e.parameter.key !== MY_SECRET_KEY) { return createJsonResponse({error: "Invalid secret key."}); } if (e.parameter.type === 'weather') { return fetchKmaWeather(e.parameter.nx, e.parameter.ny, WEATHER_API_KEY); } return fetchCalendarEvents(); } function testWeatherKey() { const apiKey = PropertiesService.getScriptProperties().getProperty('WEATHER_KEY'); const nx = "60"; const ny = "127"; console.log("--- KMA API Test Start ---"); console.log("API Key: " + apiKey); const result = fetchKmaWeather(nx, ny, apiKey); console.log("Result: " + result.getContent()); console.log("--- KMA API Test End ---"); } function fetchKmaWeather(nx, ny, apiKey) { const now = new Date(); const targetTime = new Date(now.getTime() - 40 * 60 * 1000); const dateStr = Utilities.formatDate(targetTime, "GMT+9", "yyyyMMdd"); const timeStr = Utilities.formatDate(targetTime, "GMT+9", "HH00"); const url = `http://apis.data.go.kr/1360000/VilageFcstInfoService_2.0/getUltraSrtNcst?serviceKey=${apiKey}&pageNo=1&numOfRows=10&dataType=JSON&base_date=${dateStr}&base_time=${timeStr}&nx=${nx}&ny=${ny}`; try { const response = UrlFetchApp.fetch(url, {muteHttpExceptions: true}); const content = response.getContentText(); const data = JSON.parse(content); if (data.response && data.response.body && data.response.body.items) { const items = data.response.body.items.item; const weatherData = { T1H: items.find(i => i.category === 'T1H').obsrValue, PTY: items.find(i => i.category === 'PTY').obsrValue }; return createJsonResponse(weatherData); } else { const errorMsg = data.response && data.response.header ? data.response.header.resultMsg : "Unknown response format"; return createJsonResponse({error: "KMA API Error: " + errorMsg}); } } catch(err) { return createJsonResponse({error: "KMA Fetch Failed: " + err.toString()}); } } function fetchCalendarEvents() { const now = new Date(); const nextWeek = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000); const events = CalendarApp.getDefaultCalendar().getEvents(now, nextWeek); const result = events.map(event => ({ text: event.getTitle(), time: Utilities.formatDate(event.getStartTime(), "GMT+9", "MM/dd HH:mm") })); return createJsonResponse(result); } function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
